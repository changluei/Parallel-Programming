void matrixVectorDotNaive(const vector<double> &a, const vector<double> &b, vector<double> &sum, int n) {
    for (int col = 0; col < n; ++col) {
        double value = 0.0;
        for (int row = 0; row < n; ++row) {
            value += b[static_cast<size_t>(row) * n + col] * a[row];
        }
        sum[col] = value;
    }
}

void matrixVectorDotOptimized(const vector<double> &a, const vector<double> &b, vector<double> &sum, int n) {
    fill(sum.begin(), sum.end(), 0.0);
    for (int row = 0; row < n; ++row) {
        const double factor = a[row];
        const size_t offset = static_cast<size_t>(row) * n;
        for (int col = 0; col < n; ++col) {
            sum[col] += b[offset + col] * factor;
        }
    }
}

double naiveChainSum(const vector<double> &data) {
    double sum = 0.0;
    for (double value : data) {
        sum += value;
    }
    return sum;
}

double twoWayChainSum(const vector<double> &data) {
    double sum1 = 0.0;
    double sum2 = 0.0;
    int i = 0;
    const int n = static_cast<int>(data.size());
    for (; i + 1 < n; i += 2) {
        sum1 += data[i];
        sum2 += data[i + 1];
    }
    for (; i < n; ++i) {
        sum1 += data[i];
    }
    return sum1 + sum2;
}
